mutation createFolderVar($ParentGraphQLFolderName: String!, $ChildFolderName: String!) {
  createFolder(
    repositoryIdentifier: "CONTENT"
    folderProperties: {name: $ChildFolderName, parent: {identifier: $ParentGraphQLFolderName}}
  ) {
    name
    className
    pathName
  }
}
